import {Aside} from "@astrojs/starlight/components";
import { Steps } from '@astrojs/starlight/components';

<Steps>
1. <a href="https://mcs.mail.ru/" target="blank">Перейдите в VK CLOUD</a> и создайте учетную запись

2. Перейдите на вкладку Объектное хранилище -> Бакеты
3. Нажмите на **Добавить**
4. Введите название бакета
5. Настройка ACL по умолчанию: **public-read**
6. Остальные настройки оставьте как есть и нажмите **Create a Bucket**
7. Перейдите на вкладку **Объектное хранилище** -> **Бакеты**
8. Нажмите на название бакета -> **Ключи**
9. Нажмите на **Добавить ключ**
10. Введите имя нажмите **Создать**
11. Сохраните **Access Key ID** и **Secret Key** endpoint_url и region [в личном кабинете](https://admin.defaultuploader.com)
</Steps>
<Aside title="File endpoint и Region">
  ```
  // лучше сверить с документацией провайдера
  endpoint_url: https://hb.vkcs.cloud или https://hb.ru-msk.vkcs.cloud
  region: ru-msk
  ```
</Aside>
